router.post('/', async (req,res) => {
    const posts = await loadPostCollection();

    await posts.insertOne({
        uid: req.body.uid,
        text: req.body.text,
        name: req.body.name,
        email:req.body.email,
        createdAt: new Date()
    });
    res.status(201).send(); 
});
